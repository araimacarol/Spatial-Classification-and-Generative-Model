34 std::string Token::toString() const {
35  stringstream str("#");
36  str << tokNum << " type=";
37  switch (type) {
38  case untypedT: str << "untyped"; break;
39  case charT: str << "c \'" << c << "\'"; break;
40  case intT: str << "i " << i; break;
41  case doubleT: str << "d " << d; break;
42  case stringT: str << "s \"" << s << "\""; break;
43  case boolT: str << "b " << b; break;
44  case commentT: str << "comment \"" << s << "\""; break;
45  case intRangeT: str << "range " << start << " - " << end; break;
46  }
47  return str.str();
48 }
