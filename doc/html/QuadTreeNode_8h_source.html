126  int intersects(point p, double r) {
127  // return true iff this box overlaps a circle centered at p with radius r.
128  // First check if the boundary's lie completely outside the
129  if (p.x + r < boundary.a.x) return 0;
130  if (p.x - r > boundary.b.x) return 0;
131  if (p.y + r < boundary.a.y) return 0;
132  if (p.y - r > boundary.b.y) return 0;
133 
134  // next check if corners are within range r of the centre:
135  int cornersInRange(0);
136  double r2 = r*r;
137  cornersInRange += (sqrDist(p, boundary.a) <= r2) ? 1 : 0;
138  cornersInRange += (sqrDist(p, boundary.b) <= r2) ? 1 : 0;
139  cornersInRange += (sqrDist(p.x, p.y, boundary.a.x, boundary.b.y) <= r2) ? 1 : 0;
140  cornersInRange += (sqrDist(p.x, p.y, boundary.b.x, boundary.a.y) <= r2) ? 1 : 0;
141  if (cornersInRange > 0) {
142  return cornersInRange; // XXX Still need to describe full or partial overlap
143  }
144  /*
145  * remaining possibility for a non-empty intersection is that 2 or 4 corners of the square lie
146  * *outside* the circle but *inside* the minimal square containing it:
147  */
148  if ((boundary.a.x <= p.x) && (p.x <= boundary.b.x)) return 1;
149  if ((boundary.a.y <= p.y) && (p.y <= boundary.b.y)) return 1;
150  return 0; // XXX STUB
